    classDef start fill:#F3E5F5,stroke:#7B1FA2,stroke-width:2px,color:#4A148C
    classDef process fill:#E1F5FE,stroke:#0277BD,stroke-width:2px,color:#01579B
    classDef decision fill:#FFF8E1,stroke:#F57F17,stroke-width:2px,color:#E65100
    classDef data fill:#E8F5E9,stroke:#2E7D32,stroke-width:2px,color:#1B5E20
    classDef output fill:#C8E6C9,stroke:#2E7D32,stroke-width:2px,color:#1B5E20
    classDef skip fill:#ECEFF1,stroke:#455A64,stroke-width:2px,color:#263238
    classDef apply fill:#E0F2F1,stroke:#00695C,stroke-width:2px,color:#004D40

    Trigger["Trigger
--once / --daemon (30 min)
/ session-end hook"]:::start

    Trigger --> Lock{"Acquire file lock
(PID-based staleness)"}:::decision
    Lock -->|"Locked by
another process"| Skip["Skip cycle"]:::skip
    Lock -->|"Acquired"| Load

    Load["Load CLAUDE.md
+ CLAUDE.local.md"]:::process

    Load --> FindCandidates["Find promotable local rules
Rules in local not in root,
or with changed properties"]:::process

    FindCandidates --> HasCandidates{"Candidates
found?"}:::decision
    HasCandidates -->|"None"| NoOp["Log: no promotable rules"]:::skip

    HasCandidates -->|"Yes"| BuildCandidate["Build candidate CLAUDE.md
Insert auto-promotion section
with rule metadata"]:::process

    BuildCandidate --> HashCheck{"Same hash as
last below-threshold
candidate?"}:::decision
    HashCheck -->|"Yes"| Unchanged["Skip: unchanged candidate"]:::skip

    HashCheck -->|"No"| Score["Score both versions
analyze(root) vs analyze(candidate)
benchmark(root, candidate)"]:::process

    Score --> ABCheck{"--ab flag?"}:::decision
    ABCheck -->|"Yes"| ABBench["A/B Benchmark
Synthetic executor
baseline vs guided"]:::process
    ABCheck -->|"No"| ThresholdCheck

    ABBench --> ABGate{"deltaGain >=
--min-ab-gain?"}:::decision
    ABGate -->|"No"| Proposal
    ABGate -->|"Yes"| ThresholdCheck

    ThresholdCheck{"delta >=
--min-delta?
AND --apply?"}:::decision

    ThresholdCheck -->|"No"| Proposal["Save proposal file
proposals/CLAUDE.promoted.*.md"]:::data

    ThresholdCheck -->|"Yes"| Apply["Apply promotion"]:::apply
    Apply --> Backup["Backup CLAUDE.md
backups/CLAUDE.md.*.bak"]:::data
    Apply --> Write["Write updated CLAUDE.md
with auto-promotion section"]:::data
    Apply --> ADR["Write ADR
docs/adr/ADR-NNN-*.md
with metrics + rationale"]:::data

    Proposal --> Report["Write autopilot-report.json
+ update autopilot-state.json
+ append autopilot.log"]:::output
    Apply --> Report

    Report --> Release["Release lock"]:::process
    NoOp --> Release
    Unchanged --> Release
    Skip --> Done["Done"]:::start
    Release --> Daemon{"--daemon?"}:::decision
    Daemon -->|"Yes"| Wait["Wait 30 min
then repeat"]:::process
    Wait --> Trigger
    Daemon -->|"No"| Done
